(** [match_res] is a type that indicates whether there exists a match between
    two cards or not. *)
type match_res =
  | Match of string
  | NoMatch

(** [capitalize_entry lst] is a case-sensitive translation of a parsed user 
    input [lst]. *)
let capitalize_entry lst =
  List.map (fun str -> 
    String.mapi (fun idx c -> 
        if idx = 0 then Char.uppercase_ascii c else c) str) 
    lst

(** [find_idx a num idx] is the index [idx] of a value that has a key value of 
    [num] in association array [a]. *)
let rec find_idx a num idx =
  if fst a.(idx) = num then idx
  else find_idx a num (idx + 1)

(** [lst_intersect acc src target] is a random card in intersection between 
    two card lists [src] and [target] <TODO>. 
    intersection between src list and target list (src being the guess and target being the other players' hands)*)
let rec lst_intersect acc src target =
  match src with
  | [] -> acc
  | h :: t -> begin
    if List.mem h target then lst_intersect (h :: acc) t target
    else lst_intersect acc t target
  end

(** [get_hd] is the head element of the input list. *)
let get_hd = function
  | [] -> failwith "Please file an issue"
  | h :: _ -> h

(* From the hands of other players, with intersections, (h) would have the first player that has at least one intersection.
   then randomly select one from that list *)
(** [return_random] is the random element of the first list in the nested list
    input. *)
let return_random = function
  | [] -> NoMatch
  | h :: _ -> Match (random_shuffle h |> get_hd)

(* Go through all other playsers' hands and see if any of them match with guess. If there is a match, then 
   return match_res type Match of string, where string would be the name of the card
   NoMatch if there are none. *)
(** [find_guess st_arr guess idx len] is a single random card that matches the
    guess [guess] from the hands of other players' [st_arr], except for the
    current player number [idx].

    Returns Match of string, where string would be the name of the card
      NoMatch if there are none. *)
let find_guess st_arr guess idx =
  let comparator = ref [] in
  let extract_hands =
    for i = 0 to Array.length st_arr - 1 do
      if i = idx then comparator := !comparator
      else begin
        let other_player_state = snd st_arr.(i) in
        let hands = other_player_state.hand |> unwrap_card_lst in
        comparator := hands :: !comparator
      end
    done;
    !comparator |> List.rev
  in
  List.map (fun target -> lst_intersect [] guess target) extract_hands
  |> List.filter (fun x -> x <> [])
  |> return_random

(** [print_string_list lst] prints the input string list [lst] *)
let print_string_list lst =
  (* List.iteri (fun i x -> if i = List.length lst - 1 then print_endline (string_of_int x) else print_string (string_of_int x ^ " ")) lst *)
  List.iter (fun x -> txt_c (x ^ " ")) lst

let print_int_lst lst =
  List.iter (fun x -> txt_r (string_of_int x ^ " ")) lst

(** [ids_of_user st_arr] is a list of players' locations in the map. *)
let ids_of_user st_arr =
  Array.map (fun (i, st) -> st.location.id) st_arr |> Array.to_list

(** [board_disp res st_arr cps] either displays the regular
    board with different players' locations or board with movable locations. *)
let board_disp res st_arr cps =
  if dice_sum res <> 0 then 
    print_suggestions (get_suggestions cps.location.id (dice_sum res))
  else print_game_board (ids_of_user st_arr)

(** [user_loop st i n s res] is the main game loop with 
    * [m] - map that the game is being played on
    * [st] - current player's state
    * [st_arr] - array of all players' states
    * [res] - result of rolling the dice in previous turn. 
    * [scl] - shared card list (cards remaining after distributing all the cards evenly) *)
let rec user_loop m st st_arr res scl =
  (* current player state *)
  let cps = snd st in
  (* current player number *)
  let cpn = string_of_int (fst st + 1) in
  (* curernt player's index *)
  let idx = find_idx st_arr (fst st) 0 in
  (* number of players *)
  let num_p = Array.length st_arr in
  (* print_endline (string_of_int cps.location.id); *)
  print_endline (get_suggestions cps.location.id (dice_sum res) |> List.hd |> snd |> string_of_int);
  board_disp res st_arr cps;
  print_all 0 scl res ["A did this"; "B did this"; "C did this"];
  print_endline ("Player " ^ cpn ^ "'s turn");
  print_string "> ";
  match read_line () with
  | str -> parse_str str m st st_arr res scl cps cpn idx num_p

and parse_str str m st st_arr res scl cps cpn idx num_p =
  try 
    let _ = Sys.command "clear" in 
    match parse str with
    | Roll -> roll_logic res m st st_arr scl cps
    | Go t -> go_logic t m st st_arr res scl cps idx num_p
    | Hand -> hand_logic m st st_arr scl cps
    | History -> begin
      print_game_sheet (List.map (fun (x, b) -> 
          match x with | Suspect s | Room s | Weapon s -> (s, b)) 
          cps.tracker);
      let _ = Sys.command "clear" in 
      user_loop m st st_arr (0, 0) scl
    end
    | Guess t -> guess_logic t m st st_arr scl cps idx num_p
    | Answer t -> answer_logic t m st st_arr res scl cpn idx num_p
    | Help -> begin
        print_instr "help"; 
        let _ = Sys.command "clear" in 
        user_loop m st st_arr res scl
      end 
    | Quit -> print_endline "Thank you for playing! \n"; Stdlib.exit 0
  with
  | Empty -> begin
      print_endline "Type something in! When stuck, type 'help' to get assistance.\n"; 
      user_loop m st st_arr res scl
    end
  | Malformed -> begin
      print_endline "Sorry, the command could not be understood. Type 'help' to get assistance.\n";
      user_loop m st st_arr res scl
    end

(** [roll_logic res m st st_arr scl cps] is a user routine for roll command. TODO *)
and roll_logic res m st st_arr scl cps =
  let summed = dice_sum res in
  if summed = 0 then user_loop m st st_arr (roll 6) scl
  else begin
    print_endline "You have already rolled in your turn.";
    print_endline ("You can move " ^ string_of_int summed ^ " tiles.\n");
    user_loop m st st_arr res scl
  end

and answer_logic t m st st_arr res scl cpn idx num_p =
  (* TODO: To make it more interesting, one can impose restriction that 
     "the player must be in Cellar (or just any room) to attempt to answer".*)
  let ans = unwrap_card_lst case_file in
  let attempt = capitalize_entry t in
  if List.length attempt <> 3 then begin
    print_endline "The answer takes in three inputs. Type \"Help\" if you need help.";
    user_loop m st st_arr res scl
  end
  else if ans = attempt then begin
    print_endline ("Player " ^ cpn ^ " guessed the answer correctly.");
    print_string "The answer for this game was ";
    print_card_lst case_file;
    print_endline "Thank you for playing! \n"; 
    Stdlib.exit 0
  end
  else begin
    print_endline ("Player " ^ cpn ^ " guessed the answer incorrectly.");
    print_endline "You are no longer going to be able to participate in this game.\n";
    print_endline "Press Enter to continue...";
    match read_line () with
    | _ -> begin
      (* txt_r "Uncomment in case of debugging: "; *)
      (* print_card_lst case_file; *)
      let new_st_arr = 
        List.remove_assoc (fst st) (Array.to_list st_arr) 
        |> Array.of_list 
      in
      let new_len = Array.length new_st_arr in
      if idx = num_p - 1 then
        user_loop m new_st_arr.(idx mod new_len) new_st_arr res scl
      else user_loop m new_st_arr.(idx) new_st_arr res scl
    end
  end 

and go_logic t m st st_arr res scl cps idx num_p =
  if dice_sum res = 0 then begin
    print_endline "Roll the dice first before moving using \"Roll\" command.\n";
    user_loop m st st_arr (0, 0) scl
  end
  else begin
    (* 
    Legal, Illegal needs to change. 'go' function no longer needed in player. 

    if List.length t <> 1 then 
      invalid entry; 
      query the user input again.
    else 
      1. extract out the number from the list t
      2. see if that number is indeed one of the location options (result from main2 numbered) 
      3. If it is a valid option, update the current player's location and move onto the next player
          If not, query user input again.

    *)
    (* TODO: given the tile id, figure out its description. also, *)
    try
      let new_id = int_of_string (get_hd t) in
      let movable = get_suggestions cps.location.id (dice_sum res) |> List.map (fun (i, s) -> s) in
      if not (List.mem new_id movable) then begin
        print_endline "The number must be one of the suggested outputs";
        (* print_endline (movable |> List.hd |> string_of_int); *)
        user_loop m st st_arr res scl
      end
      else begin
        let new_loc = { id = int_of_string (get_hd t); desc = "Temp" } in
        let updated_state = update_location new_loc (snd st) in
        st_arr.(idx) <- (fst st, updated_state);
        user_loop m st_arr.((idx + 1) mod num_p) st_arr (0, 0) scl
      end
    with
    | _ -> print_endline "The input must be an integer number that corresponds to the suggestions"; user_loop m st st_arr res scl
  end

and hand_logic m st st_arr scl cps = 
  (* TODO: Pretty-print the hand, like the history command *)
  let _ = Sys.command "clear" in 
  print_endline "Cards in your hand:";
  print_hands 0 cps.hand;
  print_endline "Press Enter to continue...";
  match read_line () with
  | exception End_of_file -> ()
  | _ -> let _ = Sys.command "clear" in user_loop m st st_arr (0, 0) scl

and guess_logic t m st st_arr scl cps idx num_p = 
  let guess = capitalize_entry t in
  print_string "The player asked other players to see if they have ";
  print_string_list guess;
  print_endline "cards.";
  match find_guess st_arr guess idx with
  | Match a -> begin
    let new_card = find_category a in
    let new_mark = update_tracker cps.tracker new_card in
    let new_st = update_state new_mark cps in
    st_arr.(idx) <- (fst st, new_st);
    print_endline "There was a match. One of those matches will automatically get recorded.";
    user_loop m st_arr.((idx + 1) mod num_p) st_arr (0, 0) scl
  end
  | NoMatch -> print_endline "No other players had cards that you asked for.";
    user_loop m st st_arr (0, 0) scl
