(* Single-player logic: only given a default of 3 cards.
   need to deduce answer by answering a question when in a room tile.
   one cannot answer twice on the same room. either go back out and come back in or go to different location. 

   once landed in the room, the player can either take a guess on the answer or take hints (getting information about one
   card that is not in the player's or computer's hand, and is not the answer.).
   computer will always attempt to answer.
*)

(** [prob_det p] is true if the random float number is less than the
    probability value [p] and false otherwise. *)
let prob_det p =
  let () = Random.self_init () in
  if Random.float 1.0 < p then true
  else false

(* TODO: this is in users actually.. *)
(** [get_hd] is the head element of the input list. *)
let get_hd = function
  | [] -> failwith "Please file an issue"
  | h :: _ -> h

(* TODO: will be removed later since it is a duplicate of the one in main. *)
(** [capitalize_entry lst] is a case-sensitive translation of a parsed user 
    input [lst]. *)
let capitalize_entry lst =
  List.map (fun str -> 
      String.mapi (fun idx c -> 
          if idx = 0 then Char.uppercase_ascii c else c) str) 
    lst

let rec count_tracker acc = function
  | [] -> acc
  | (_, tf) :: t -> begin
      if not tf then count_tracker (acc + 1) t 
      else count_tracker acc t
    end

let ready_to_answer st_cpu =
  let count_false = count_tracker 0 st_cpu.tracker in
  if count_false = 3 then true
  else false

let rec find_missing acc = function
  | [] -> acc |> List.rev
  | (c, tf) :: t -> begin
      if not tf then find_missing (c :: acc) t
      else find_missing acc t
    end

let ai_routine intro p st_cpu =
  print_endline intro;
  (* Dice roll result for CPU *)
  let res = dice_sum (roll 6) in
  (* CPU's current location *)
  let cpu_loc = st_cpu.location.desc in
  (* List of tiles that the CPU can move with dice result 'res' *)
  let movable = get_suggestions st_cpu.location.id res in
  (* Next tile ID that the CPU is going to move to. *)
  let next_loc = get_hd (random_shuffle movable) in
  (* Something that finds the description of the tile that corresponds to the id 'next_loc', call it tile_name *)
  let tile_name = desc_of_id (snd next_loc) in
  (* Room names is a list of room names that are in play in this game *)
  let room_names = ["Kitchen"; "Utility"; "Laundry"; "Dining"; "Billiard"; "Library"; "Lounge"; "Hall"; "Attic"] in
  (* Probabilistically determine whether to track the guess or not. *)
  let guess_and_track = prob_det p in

  print_endline ("Computer is at " ^ cpu_loc);
  print_endline ("Computer rolled a dice to move " ^ (string_of_int res) ^ " tiles.");
  (* next_loc would be an int -> need to find what description that number tile has. *)
  print_endline ("Computer will move to a tile with id " ^ (string_of_int (snd next_loc)));
  (* Here, print the current map *)

  let new_tile = {id = snd next_loc; desc = tile_name} in

  if List.mem tile_name room_names then begin
    if tile_name = cpu_loc then update_location new_tile st_cpu
    else begin
      let card_here = find_missing [] st_cpu.tracker |> random_shuffle in
      if guess_and_track then begin
        print_endline "If guess_and_track is set, track one of the three cards if none match.";
        let mark = update_tracker st_cpu.tracker (List.hd card_here) in
        update_state mark st_cpu |> update_location new_tile
      end
      else update_location new_tile st_cpu
    end
  end
  else update_location new_tile st_cpu

(** [move_ai_easy intro st_cpu] is a routine for medium-level difficulty AI,
    where it keeps track of cards it matched with the answer with 25% chance. *)
let move_ai_easy intro st_cpu = 
  ai_routine intro 0.25 st_cpu

(** [move_ai_medium intro st_cpu] is a routine for medium-level difficulty AI,
    where it keeps track of cards it matched with the answer with 50% chance. *)
let move_ai_medium intro st_cpu =
  ai_routine intro 0.50 st_cpu

(** [move_ai_hard intro st_cpu] is a routine for hard-level difficulty AI, 
    where it keeps track of the cards it matched with the answer with 75% 
    chance. *)
let move_ai_hard intro st_cpu =
  ai_routine intro 0.75 st_cpu

(** [is_ai_turn] is true if and only if it is AI's turn to move. *)
let is_ai_turn = ref false

let rec single_player_loop m diff st st_cpu res =
  if !is_ai_turn then begin
    (* TODO: fix *)
    let new_cpu_st = diff "Computer's turn" st_cpu in
    is_ai_turn := false;
    print_endline "Press Enter to continue...";
    match read_line () with
    | exception End_of_file -> ()
    | _ -> single_player_loop m diff st new_cpu_st res
  end
  else begin
    print_game_board [st.location.id];
    print_endline "\nYour turn";
    print_string "> ";
    match read_line () with
    | str -> begin
        try
          let _ = Sys.command "clear" in 
          match parse str with
          | Roll -> begin
              if res = 0 then
                let r = roll 6 in
                print_endline ("(" ^ string_of_int (fst r) ^ ", " ^ string_of_int (snd r) ^ ")\n");
                single_player_loop m diff st st_cpu (dice_sum r)
              else begin
                print_endline ("You have already rolled. You can move " ^ string_of_int res ^ " tiles.\n");
                single_player_loop m diff st st_cpu res
              end
            end
          | Go t -> begin
              is_ai_turn := true;
              if res = 0 then begin
                print_endline "Roll the dice first before moving using \"Roll\" command.\n";
                single_player_loop m diff st st_cpu res
              end
              else begin
                (* if List.length t <> 1 then begin
                   print_endline "Type one of the options given in the suggestions";
                   single_player_loop m diff st st_cpu res
                   end *)
                (* else begin *)
                (* TODO: given the tile id, figure out its description. also, *)
                try
                  (* TODO: here, check that the number is indeed a valid number in the suggestion list. *)
                  let new_room_id = int_of_string (get_hd t) in
                  let new_loc = { id = new_room_id; desc = desc_of_id new_room_id } in
                  let updated_state = update_location new_loc st in
                  is_ai_turn := true;
                  single_player_loop m diff updated_state st_cpu res
                with
                | _ -> print_endline "The input must be an integer number that corresponds to the suggestions"; 
                  single_player_loop m diff st st_cpu res
              end
              (* end *)
            end
          | Hand -> begin
              let _ = Sys.command "clear" in 
              print_endline "Cards in your hand:";
              print_hands 0 st.hand;
              print_endline "Press Enter to continue...";
              match read_line () with
              | exception End_of_file -> ()
              | _ -> let _ = Sys.command "clear" in single_player_loop m diff st st_cpu res
            end
          | History -> begin
              print_game_sheet (List.map (fun (x, b) -> 
                  match x with 
                  | Suspect s | Room s | Weapon s -> (s, b)) 
                  st.tracker);
              let _ = Sys.command "clear" in 
              single_player_loop m diff st st_cpu res
            end
          (* User asks the CPU to show his or her card *)
          | Guess t -> begin
              is_ai_turn := true;
              print_endline "The player asked other players to show <><><> cards.";
              single_player_loop m diff st st_cpu res
            end 
          | Answer t -> begin
              (* TODO: To make it more interesting, one can impose restriction that "the player must be in Cellar to attempt to answer".*)
              let ans = unwrap_card_lst case_file in
              let attempt = capitalize_entry t in
              if List.length attempt <> 3 then begin
                print_endline "The answer takes in three inputs. Type \"Help\" if you need help.";
                single_player_loop m diff st st_cpu res
              end
              else if ans = attempt then begin
                print_endline "You guessed the answer correctly!";
                print_string "The answer for this game was ";
                print_card_lst case_file;
                print_endline "Thank you for playing! \n"; 
                Stdlib.exit 0
              end
              else begin
                is_ai_turn := true;
                print_endline "Your answer attempt was incorrect!";
                single_player_loop m diff st st_cpu res
              end
            end
          | Help -> begin
              print_endline "Display help instruction for single-player";
              match read_line () with
              | exception End_of_file -> ()
              | _ -> let _ = Sys.command "clear" in 
                single_player_loop m diff st st_cpu res
            end 
          | Quit -> print_endline "Thank you for playing! \n"; Stdlib.exit 0
        with
        | Empty -> begin
            print_endline "Type something in! When stuck, type 'help' to get assistance.\n"; 
            single_player_loop m diff st st_cpu res
          end
        | Malformed -> begin
            print_endline "Sorry, the command could not be understood. Type 'help' to get assistance.\n";
            single_player_loop m diff st st_cpu res
          end
      end
  end
