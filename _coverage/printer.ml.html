(** [print_game_sheet card] prints the game sheet to the player. *)
let print_game_sheet cards = 
  let _ = Sys.command "clear" in
  let line = "|____________________|__________________|\n" in
  let space = "|                    |                  |\n" in
  txt_c "                GAME SHEET\n";
  print_string " _______________________________________\n";
  let rec cards_sheet = function
    | [] -> print_string line;
    | h :: t -> print_string (line ^ space); 
      if snd h then 
        print_string ("|\t" ^ fst h ^ "\t     |\t\tO\t|\n")
      else 
        print_string ("|\t" ^ fst h ^ "\t     |\t\tX\t|\n");
      cards_sheet t 
  in 
  cards_sheet cards;
  print_endline "Enter any key to continue...";
  match read_line () with
  | exception End_of_file -> ()
  | _ -> ()

  if room <> loc then print_string ("|\t" ^ room ^ "\t\t")
  else begin
    print_string "|\t";
    txt_y loc;
    print_string "\t\t";
  end

let unwrap_card_lst card_lst =
  List.map (fun c -> 
      match c with | Suspect s | Room s | Weapon s -> s) 
    card_lst

let print_card_lst lst =
  let cl = unwrap_card_lst lst in
  let rec print_card_lst_aux = function
    | [] -> print_endline "\n";
    | h :: t -> begin
        print_string (h ^ " | ");
        print_card_lst_aux t
      end
  in
  print_card_lst_aux cl

(** [repeat n str] prints the string [str] for [n] iterations. *)
let rec repeat n str = 
  if n = 0 then str
  else repeat (n - 1) str ^ str

(** [title n str] prints the TODO *)
let title n str = 
  let t = repeat n "----------------" in
  let s = repeat n "            " in
  if n + 1 <= 1 then 
    let t = "---------------  " in
    let s = "       " in
    ("\n\n" ^ str ^ s, t)
  else if n + 1 = 3 then 
    let s = repeat n "             " in
    ("\n\n" ^ str ^ s ^ "  ", t ^ "-  ")
  else ("\n\n" ^ str ^ s, t ^ "--")

(** [space len str] *)
let rec space len str =
  if len = 0 then str
  else space (len - 1) str ^ " "

(** [faces num] prints a face of one die with the outcome [num]. *)
let faces num =
  let empty = "|         |" in
  let mid =  "|    o    |" in
  let double = "|   o o   |" in
  let left = "|   o     |" in
  let right = "|     o   |" in
  if num = 1 then [empty; mid; empty]
  else if num = 2 then [left; empty; right]
  else if num = 3 then [left; mid; right]
  else if num = 4 then [double; empty; double]
  else if num = 5 then [double; mid; double]
  else if num = 6 then [double; double; double]
  else [empty; empty; empty]

(** [print_dots dice] prints the face-up value of the dice [dice] that is 
    rolled. *)
let print_dots dice = 
  let f a b = (a, b) in
  List.map2 f (faces (fst dice)) (faces (snd dice))

(** [spaces suspect] TODO *)
let spaces suspect =
  let rec spaces_acc suspect acc = 
    match suspect with
    | [] -> List.rev acc
    | h :: t -> begin
        spaces_acc t ([""; h] @ acc)
      end 
  in 
  spaces_acc suspect []

  let len = List.length suspect in
  if r = 0 then txt_g "SUSPECTS                  "
  else if r = 1 then txt_g "------------------------"
  else if r = 2 then print_string " "
  else if r < len + 3 then begin
    let rec print_spaces i suspect = 
      match suspect with
      | [] -> ()
      | h :: t -> 
        if h = "" then print_string " "
        else if i = r - 3 then begin
          print_string "* ";
          txt_g h;
        end
        else print_spaces (i + 1) t
    in 
    print_spaces 0 suspect
  end
  else ()
